// THE AI BRAIN window.askAI = async () => { const prompt = document.getElementById('ai-prompt').value; const resBox = document.getElementById('ai-response'); if(!prompt) return; resBox.style.display = 'block'; resBox.innerText = "🤖 Thinking..."; try { // I pulled the key into its own variable to make the URL cleaner const apiKey = "AIzaSyDL3b0X2ZWoolhU4g1wLNu4GcevpN2xqMg"; const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${apiKey}`; const response = await fetch(url, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ contents: [{ parts: [{ text: `You are an IEP Assistant for KIPP Navigate. Use strengths-based, professional language. Request: ${prompt}` }] }] }) }); const data = await response.json(); // This checks if Google sent back an error message instead of an answer if (data.error) { resBox.innerText = "❌ Google AI Error: " + data.error.message; } else if (data.candidates && data.candidates[0].content.parts[0].text) { resBox.innerText = data.candidates[0].content.parts[0].text; } else { resBox.innerText = "❌ AI responded but the format was unexpected."; } } catch (e) { resBox.innerText = "❌ Network Error: Could not reach AI. Check your internet or API key."; console.error(e); } };
